static inline int32_t float_to_q31(float x)
{
    if (x >  0.999999f) x =  0.999999f;
    if (x < -1.0f)      x = -1.0f;
    return (int32_t)(x * 2147483648.0f); // 2^31
}

static inline float cordic_atan2_rad(float y, float x){
    // 0,0 예외
//    float ax = fabsf(x), ay = fabsf(y);
//    float m = (ax > ay) ? ax : ay;
//    if (m < 1e-12f) return 0.0f;

    float xn = x / measure_limit;
    float yn = y / measure_limit;

    int32_t Xq = float_to_q31(xn);
    int32_t Yq = float_to_q31(yn);

    // 입력 write (X, Y)
    LL_CORDIC_WriteData(CORDIC, (uint32_t)Xq);
    LL_CORDIC_WriteData(CORDIC, (uint32_t)Yq);

    // 결과 read (각도)
    int32_t ang_q = (int32_t)LL_CORDIC_ReadData(CORDIC);

    const float Q31_TO_RAD = (float)M_PI;
    float ang = ((float)ang_q / 2147483648.0f) * Q31_TO_RAD;
    return ang;
}

static inline float wrap_pm_pi(float x){
    while (x >= PI)  x -= 2.0f * PI;
    while (x < -PI)  x += 2.0f * PI;

// ========= Common PLL (20kHz) =========
#define PLL_FS          (20000.0f)
#define PLL_TS          (1.0f / PLL_FS)
#define PLL_ZETA        (0.707f)

// wn scheduling (Hz) : 관절용 시작값 추천
#define PLL_FN_LOW_HZ   (200.0f)     // 저속(HFI 우세)에서 느리게
#define PLL_FN_HIGH_HZ  (500.0f)    // 중속~(EMF 우세)에서 빠르게

#define PLL_WN_LOW      (2.0f * (float)M_PI * PLL_FN_LOW_HZ)
#define PLL_WN_HIGH     (2.0f * (float)M_PI * PLL_FN_HIGH_HZ)

// smoothing for wn (few ms)
#define PLL_WN_ALPHA    (0.8f)

// weight schedule in electrical rad/s
#define W_EMF_LOW       (200.0f)    // 이 아래에서는 EMF 신뢰 낮음
#define W_EMF_HIGH      (400.0f)   // 이 이상에서는 EMF 위주

// public outputs (share with your code)
static float th_e_hat_wrapped = 0.0f;   // wrapped electrical angle [rad]
static float we_hat_rads      = 0.0f;   // electrical speed [rad/s]
static float th_e_cont        = 0.0f;   // continuous electrical angle [rad]

// mech shared
#define w_m_alpha (0.02f)

// internal
static uint8_t pll_inited = 0;
static float integ_e = 0.0f;
static float th_e_hat_prev = 0.0f;
static float wn_f = PLL_WN_LOW;
static float w_m_lpf_prev = 0.0f;

static inline float weight_emf_from_we(float we_abs){
    float w = (we_abs - W_EMF_LOW) / (W_EMF_HIGH - W_EMF_LOW);
    return clampf(w, 0.0f, 1.0f);
}

/**
 * @brief Common PLL update using both HFI & EMF angle measurements.
 *
 * Inputs:
 *  - th_hfi_meas : electrical angle from HFI (wrapped, rad)
 *  - th_emf_meas : electrical angle from EMF (wrapped, rad)  (already -pi/2 compensated!)
 *
 * Notes:
 *  - Call at 20kHz (every ADC callback after currents/observers updated).
 *  - It blends errors, not angles.
 *  - wn is scheduled by speed/weight, smoothed.
 */
static inline float PosEst_PLL_Update_Common(float th_hfi_meas, float th_emf_meas){
    if(!pll_inited){
        // initialize with HFI (usually more reliable at standstill)
        th_e_hat_wrapped = th_hfi_meas;
        th_e_hat_prev    = th_e_hat_wrapped;
        th_e_cont        = th_hfi_meas;
        we_hat_rads      = 0.0f;
        integ_e          = 0.0f;

        th_m_est_rad     = th_e_cont / polepair;
        w_m_est_rads     = 0.0f;

        wn_f             = PLL_WN_LOW;
        w_m_lpf_prev     = 0.0f;

        pll_inited = 1;
        return th_e_hat_wrapped;
    }

    // 1) weights
    float we_abs = fabsf(we_hat_rads);
    float w_emf  = weight_emf_from_we(we_abs);
    float w_hfi  = 1.0f - w_emf;

    // 2) phase detector (blend errors, NOT angles)
    float e_hfi = wrap_pm_pi(th_hfi_meas - th_e_hat_wrapped);
    float e_emf = wrap_pm_pi(th_emf_meas - th_e_hat_wrapped);
    float e     = w_hfi * e_hfi + w_emf * e_emf;

    // 3) schedule PLL bandwidth (wn) using w_emf
    float wn_tgt = (1.0f - w_emf) * PLL_WN_LOW + w_emf * PLL_WN_HIGH;
    wn_f = PLL_WN_ALPHA * wn_f + (1.0f - PLL_WN_ALPHA) * wn_tgt;

    float L1 = 2.0f * PLL_ZETA * wn_f;     // [1/s]
    float L2 = wn_f * wn_f;                // [1/s^2]

    // 4) integrate phase error
    integ_e += e * PLL_TS;

    // 5) PI -> electrical speed
    we_hat_rads = (L1 * e) + (L2 * integ_e);

    // optional speed clamp to avoid crazy spikes during bring-up
    // we_hat_rads = clampf(we_hat_rads, -8000.0f, 8000.0f);

    // 6) integrate to wrapped angle
    th_e_hat_wrapped = wrap_pm_pi(th_e_hat_wrapped + we_hat_rads * PLL_TS);

    // 7) continuous electrical angle build
    float dth = wrap_pm_pi(th_e_hat_wrapped - th_e_hat_prev);
    th_e_hat_prev = th_e_hat_wrapped;
    th_e_cont += dth;

    // 8) mech angle & speed (shared)
    float th_m = th_e_cont / polepair;
    float w_m  = we_hat_rads / polepair;

    // LPF mech speed for D-term stability
    w_m = w_m_alpha * w_m + (1.0f - w_m_alpha) * w_m_lpf_prev;
    w_m_lpf_prev = w_m;

    th_m_est_rad = th_m;
    w_m_est_rads = w_m;

    return th_e_hat_wrapped;
}

#define current_const 0.00805664*2  // 10V/V Op-amp gain
uint8_t first = 1;
float get_flip_start, get_flip_end, get_flip_start_prev, get_flip_end_prev;
#define flip_alpha 0.5

#define Ts (0.00005f)
#define Rs (0.765f)
#define Ls (0.0002102f)  // LCR meter phase to phase is 335.5uH ~ 505.3uH at 10k measure
#define R_over_L (Rs / Ls)
#define invL (1/Ls)

float i_hat_alpha = 0.0, i_hat_beta = 0.0;
float e_hat_alpha = 0.0, e_hat_beta = 0.0;

#define WO_MIN    (200.0f)
#define WO_MAX    (3000.0f)
#define WO_ALPHA  (0.99f)
#define ZETA      (0.707f)

static float wo_f = WO_MIN;

void EMF_observer_update(){
    // 전류는 이미 (i + i_prev)/2 로 들어오는 i_alpha_LPF/i_beta_LPF 사용
    float s_alpha = i_alpha_LPF - i_hat_alpha;
    float s_beta  = i_beta_LPF  - i_hat_beta;

    // 1) bandwidth schedule + smoothing
    float wo_tgt = 4.0f * fabsf(we_hat_rads);
    wo_tgt = clampf(wo_tgt, WO_MIN, WO_MAX);
    wo_f = WO_ALPHA * wo_f + (1.0f - WO_ALPHA) * wo_tgt;

    // 2) gains
    float k_i = 2.0f * ZETA * wo_f - R_over_L;
    float k_e = Ls * wo_f * wo_f;

    // 3) update (use previous applied voltage!)
    i_hat_alpha += Ts * ( -R_over_L * i_hat_alpha + invL*V_alpha - invL*e_hat_alpha + k_i*s_alpha );
    i_hat_beta  += Ts * ( -R_over_L * i_hat_beta  + invL*V_beta  - invL*e_hat_beta  + k_i*s_beta  );

    e_hat_alpha += Ts * ( k_e * s_alpha );
    e_hat_beta  += Ts * ( k_e * s_beta  );

    // 디버그/폭주 방지용(가능하면 0.9*Vdc로)
    // e_hat_alpha = clampf(e_hat_alpha, -ECLAMP, ECLAMP);
    // e_hat_beta  = clampf(e_hat_beta,  -ECLAMP, ECLAMP);
}

uint8_t flip_flag = 1;

void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef *hadc){  //ADC conversion complete callback, spend 5.2us to conversion
	if      (hadc->Instance == ADC1) done |= (1<<0);
	else if (hadc->Instance == ADC2) done |= (1<<1);
	else if (hadc->Instance == ADC3) done |= (1<<2);

	if(done == 7){
		done = 0;

		// alpha beta current calculation
		ic = current_A[0] - current_A_calibrated;  // A<->C mapping hardware issue
		ib = current_B[0] - current_B_calibrated;
		//ia = current_C[0] - current_C_calibrated;  // A<->C mapping hardware issue
		ia = -(ib+ic); // ia is very unstable. currnet_C has big jitter (+-10 LSB) (maybe hardware PCB problem. maybe charge pump?)

		// inv clarke
		i_alpha = ia * current_const;                   // i_alpha = 2/3 * (ia - (ib + ic)/2) = ia
		i_beta  = (ib - ic) * invsqrt3 * current_const; // i_beta  = 2/3 * (ib - ic) * sqrt(3)/2 = (ib - ic) * 1/sqrt(3)

		// digital LPF(mean sum)
		i_alpha_LPF = (i_alpha + i_alpha_prev) * 0.5;
		i_beta_LPF  = (i_beta  + i_beta_prev)  * 0.5;
		i_alpha_prev = i_alpha;
		i_beta_prev  = i_beta;

		// back-EMF start
		EMF_observer_update();

		theta_e_est_rad_EMF = wrap_pm_pi(cordic_atan2_rad(e_hat_beta, e_hat_alpha) - 0.5f*PI);

		// HFI start
		if(HFI_test){
			// rotor angle estimation with HFI
			if(have_prev_HFI){
				delta_i_alpha = i_alpha - i_alpha_prev_HFI;
				delta_i_beta  = i_beta  - i_beta_prev_HFI;

				if(V_injected < 0){
					delta_i_alpha = - delta_i_alpha;
					delta_i_beta  = - delta_i_beta;
				}

				//theta_e_est_rad = atan2f(delta_i_beta, delta_i_alpha);
				theta_e_est_rad = cordic_atan2_rad(delta_i_beta, delta_i_alpha);

				if(have_prev_estimation){
					float case1 = theta_e_est_rad;
					float case2 = theta_e_est_rad + PI;

					if(fabsf(wrap_pm_pi(case1 - theta_e_est_rad_prev)) > fabsf(wrap_pm_pi(case2 - theta_e_est_rad_prev))){
						//theta_e_est_rad = case2;
						angle_flip = 1;
					}
					else{
						//theta_e_est_rad = case1;
						angle_flip = 0;
					}
				}

				theta_e_est_rad_PLL = PosEst_PLL_Update_Common(theta_e_est_rad, theta_e_est_rad_EMF);

				theta_e_est_rad_prev = theta_e_est_rad_PLL;

				have_prev_estimation = 1;

				estimated_electrical_angle = theta_e_est_rad_PLL * (180.0f / PI);
				theta_e_wrapped_rad = theta_e_est_rad_PLL;
			}

			i_alpha_prev_HFI = i_alpha;
			i_beta_prev_HFI  = i_beta;
			have_prev_HFI    = 1;

			arm_sin_cos_f32(estimated_electrical_angle,&_sin,&_cos);

			if(count < 19999){ //estimate start, wait estimation converge
				V_alpha = 1.0f + V_injected * _cos;
				V_beta =  0.0f + V_injected * _sin;
				count++;
			}
			/*else if(count < 79999){ // for torque control, get axis flip or not
				if(count < 40000){
					get_flip_start = theta_e_est_rad_PLL * flip_alpha + (1 - flip_alpha) * get_flip_start_prev;
					get_flip_start_prev = get_flip_start;

					V_alpha = 1.0f + V_injected * _cos;
					V_beta =  0.0f + V_injected * _sin;
				}
				else if(count < 70000){
					get_flip_end = theta_e_est_rad_PLL * flip_alpha + (1 - flip_alpha) * get_flip_end_prev;
					get_flip_end_prev = get_flip_end;

					V_beta =  1.0f + V_injected * _sin;
				}
				else{
					if((get_flip_end - get_flip_start) > 0)
						flip_sign = -1;
					else
						flip_sign = 1;

					V_alpha = V_injected * _cos;
					V_beta  = V_injected * _sin;
				}
				count++;
			}*/
			else if(count == 20000){
				we_hat_rads = 0.0;
				pos_ref_mech_rad = 0.0;
				th_e_cont = 0;
				V_alpha = 1.0f + V_injected * _cos;
				V_beta =  0.0f + V_injected * _sin;
				count++;
			}
			else{
				if(count < 35000){
					if(flip_flag){
						if(w_m_est_rads > 30.0f){
							flip_sign = -flip_sign;
							flip_flag = 0;
						}
						else if(w_m_est_rads < -30.0f){
							flip_sign = -flip_sign;
							flip_flag = 0;
						}
					}
					count++;
				}

				// inv park, using high frequency rejected
				id =  i_alpha_LPF * _cos + i_beta_LPF * _sin;
				iq = -i_alpha_LPF * _sin + i_beta_LPF * _cos;

				//PI
				iq_err = iq_ref - iq;
				id_err = id_ref - id;
				iq_err_integral = iq_err_integral + iq_err;
				id_err_integral = id_err_integral + id_err;

				// Vq, Vd calculation
				float Vq_unsat = (P * iq_err) + (I * iq_err_integral);
				float Vd_unsat = (P * id_err) + (I * id_err_integral) + V_injected; // d axis injection

				Vd = (Vd_unsat > Vd_limit) ? Vd_limit : (Vd_unsat < -Vd_limit) ? -Vd_limit : Vd_unsat;

				float Vq_limit;
				arm_sqrt_f32((V_LIMIT*V_LIMIT) - (Vd*Vd), &Vq_limit);
				Vq_limit = Vq_limit * 0.9; // 0.9 for leave injection voltage
				Vq = (Vq_unsat > Vq_limit) ? Vq_limit : (Vq_unsat < -Vq_limit) ? -Vq_limit : Vq_unsat;

				iq_err_integral = iq_err_integral - Ka*(Vq_unsat - Vq);
				id_err_integral = id_err_integral - Ka*(Vd_unsat - Vd);

				// park
				V_alpha = Vd * _cos - Vq * _sin;
				V_beta  = Vd * _sin + Vq * _cos;
			}

			SVPWM(V_alpha, V_beta);

			V_injected = -V_injected; // injection with half of sampling freq

			HAL_GPIO_WritePin(GPIOC, GPIO_PIN_4, 0);
			//uint32_t debug_time_end = dwt_now();
			//real_us = cycles_to_us(debug_time_end - debug_time_start);
		}
		else{
			SVPWM(0.0,0.0);
		}
	}
	return;
}


void set_ccr(uint16_t a, uint16_t b, uint16_t c){
    TIM1->CCR1 = a; TIM1->CCR2 = b; TIM1->CCR3 = c;

static inline uint16_t clamp_u16_i32(int32_t x, uint16_t lo, uint16_t hi){
    if (x < (int32_t)lo) return lo;
    if (x > (int32_t)hi) return hi;
    return (uint16_t)x;
}

static inline int8_t sign_hyst(float x, float eps){
    if (x >  eps) return +1;
    if (x < -eps) return -1;
    return 0;
}

#define dt_ticks    	  (16.8f)
#define I_EPS             (0.05f)
#define SQRT3_BY_2        (0.8660254037844386f)
static int8_t sa_last=0, sb_last=0, sc_last=0;

static inline int8_t sign_hold(float x, float eps, int8_t *last){
    if (x >  eps) { *last = +1; return +1; }
    if (x < -eps) { *last = -1; return -1; }
    return *last; // deadband 안에서는 직전 부호 유지
}

void SVPWM(float Valpha, float Vbeta){
    float Vmag;
    arm_sqrt_f32(Valpha*Valpha + Vbeta*Vbeta, &Vmag);

    if (Vmag < 1e-6f) {
        set_ccr(ARR_TICKS, ARR_TICKS, ARR_TICKS);
        return;
    }

    float M = 1.73205080f * (Vmag / V_BUS);
    if (M > 0.99f) M = 0.99f;

    float ang = cordic_atan2_rad(Vbeta, Valpha);
    if (ang < 0) ang += 6.2831853f;

    uint8_t k = (uint8_t)(ang / SECTOR_RAD);
    if (k >= 6) k = 0;

    float phi = ang - k*SECTOR_RAD;

    float T1 = M * arm_sin_f32(SECTOR_RAD - phi);
    float T2 = M * arm_sin_f32(phi);

    int32_t t1 = (int32_t)(ARR_TICKS * T1);
    int32_t t2 = (int32_t)(ARR_TICKS * T2);

    int32_t t0 = (ARR_TICKS - t1 - t2) >> 1;
    if (t0 < 0) t0 = 0;

    uint16_t Ua, Ub, Uc;
    switch (k){
        case 0: Ua=t1+t2+t0; Ub=t2+t0;     Uc=t0;            break;
        case 1: Ua=t1+t0;    Ub=t1+t2+t0;  Uc=t0;            break;
        case 2: Ua=t0;       Ub=t1+t2+t0;  Uc=t2+t0;         break;
        case 3: Ua=t0;       Ub=t1+t0;     Uc=t1+t2+t0;      break;
        case 4: Ua=t2+t0;    Ub=t0;        Uc=t1+t2+t0;      break;
        default:Ua=t1+t2+t0; Ub=t0;        Uc=t1+t0;         break;
    }

    // deadtime compensation
    float ia_deadtime = i_alpha_LPF;
    float ib_deadtime = -0.5f * i_alpha_LPF + SQRT3_BY_2 * i_beta_LPF;
    float ic_deadtime = -0.5f * i_alpha_LPF - SQRT3_BY_2 * i_beta_LPF;

    int8_t sa = sign_hold(ia_deadtime, I_EPS, &sa_last);
    int8_t sb = sign_hold(ib_deadtime, I_EPS, &sb_last);
    int8_t sc = sign_hold(ic_deadtime, I_EPS, &sc_last);
    int32_t duA = (int32_t) ((float)sa * dt_ticks);
    int32_t duB = (int32_t) ((float)sb * dt_ticks);
    int32_t duC = (int32_t) ((float)sc * dt_ticks);

    int32_t duM = (duA + duB + duC) / 3;
    duA -= duM; duB -= duM; duC -= duM;

    int32_t Ua_c = (int32_t)Ua + duA;
    int32_t Ub_c = (int32_t)Ub + duB;
    int32_t Uc_c = (int32_t)Uc + duC;

    int32_t minU = Ua_c;
    if (Ub_c < minU) minU = Ub_c;
    if (Uc_c < minU) minU = Uc_c;
    int32_t maxU = Ua_c;
    if (Ub_c > maxU) maxU = Ub_c;
    if (Uc_c > maxU) maxU = Uc_c;
    if (minU < 0) {
        Ua_c -= minU; Ub_c -= minU; Uc_c -= minU;
        maxU -= minU;
    }
    if (maxU > ARR_TICKS) {
        int32_t sh = maxU - ARR_TICKS;
        Ua_c -= sh; Ub_c -= sh; Uc_c -= sh;
    }
    Ua = clamp_u16_i32(Ua_c, 0, ARR_TICKS);
    Ub = clamp_u16_i32(Ub_c, 0, ARR_TICKS);
    Uc = clamp_u16_i32(Uc_c, 0, ARR_TICKS);
    set_ccr(ARR_TICKS - Ua, ARR_TICKS - Ub, ARR_TICKS - Uc);
}
